# CustomDatasourceConfig

Structure describing config properties of a custom datasource

## Example Usage

```typescript
import { CustomDatasourceConfig } from "@gleanwork/api-client/models/components";

let value: CustomDatasourceConfig = {
  name: "<value>",
  urlRegex: "https://example-company.datasource.com/.*",
  quicklinks: [
    {
      iconConfig: {
        color: "#343CED",
        key: "person_icon",
        iconType: "GLYPH",
        name: "user",
      },
    },
  ],
};
```
