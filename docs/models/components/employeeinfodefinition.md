# EmployeeInfoDefinition

Describes employee info

## Example Usage

```typescript
import { EmployeeInfoDefinition } from "@gleanwork/api-client/models/components";

let value: EmployeeInfoDefinition = {
  email: "Jocelyn.Hagenes@hotmail.com",
  department: "<value>",
  datasourceProfiles: [
    {
      datasource: "github",
      handle: "<value>",
    },
  ],
};
```
