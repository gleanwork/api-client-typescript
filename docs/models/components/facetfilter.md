# FacetFilter

## Example Usage

```typescript
import { FacetFilter } from "@gleanwork/api-client/models/components";

let value: FacetFilter = {
  fieldName: "type",
  values: [
    {
      value: "Spreadsheet",
      relationType: "EQUALS",
    },
    {
      value: "Presentation",
      relationType: "EQUALS",
    },
  ],
};
```
