# Shortcut

## Example Usage

```typescript
import { Shortcut } from "@gleanwork/api-client/models/components";

let value: Shortcut = {
  inputAlias: "<value>",
  destinationUrl: "https://second-airmail.org",
  createdBy: "<value>",
};
```
